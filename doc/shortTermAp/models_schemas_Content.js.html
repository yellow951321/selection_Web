    allowNull: true,

  title3: {
    type: Sequelize.STRING(200),
    allowNull: true,
  },
  title4: {
    type: Sequelize.STRING(200),
    allowNull: true,
  },
  content: {
    type: Sequelize.STRING(1000),
    allowNull: true,
  },
  summary: {
    type: Sequelize.STRING(200),
    allowNull: true,

  note: {
    type: Sequelize.STRING(1000),
    allowNull: true,

  pageFrom: {
